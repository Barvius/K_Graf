#include "stdafx.h"
#include <glut.h>
#include <math.h>

void display() {
	glClear(GL_COLOR_BUFFER_BIT); // очистка окна цветом фона

	glBegin(GL_LINE_STRIP);
	for (GLfloat x = -10.0; x < 10.0; x += 0.1){
		GLfloat y = (1.85 - x) * cos(3.5 * x - 0.5);
		glVertex2f(x, y);
		printf("%f\t%f\n",x,y);
	}
	glEnd();

	glColor3f(1.0f, 0.0f, 0.0f);
	glBegin(GL_LINES);
	glVertex2f(-10,0);
	glVertex2f(10, 0);
	glVertex2f(0, -15);
	glVertex2f(0, 15);
	glEnd();

	glFlush(); // отправляем весь вывод на дисплей
}

void Start() {
	glClearColor(0, 0, 0, 0);
	glMatrixMode(GL_PROJECTION);
	glLoadIdentity();
	glOrtho(-10, 10, -12, 12, -1, 1);
	glClearColor(1.0, 1.0, 1.0, 1.0);
	glColor3f(0.0f, 0.0f, 0.0f);
	glPointSize(0.1);
}

int main(int argc, char** argv) {
	glutInit(&argc, argv);
	glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
	glutInitWindowSize(1280, 690);
	glutInitWindowPosition(0, 0);
	glutCreateWindow("Rownanie");
	Start();
	glutDisplayFunc(display);
	glutMainLoop();
	return 0;
}
